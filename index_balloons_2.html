function makeBalancedCounts(total){ const n = COLORS.length; // 8 couleurs let gap; if(total < 160){ gap = 4; }else if(total < 260){ gap = 3; }else{ gap = 2; } const staircase = gap * (n * (n - 1) / 2); let base = Math.floor((total - staircase) / n); if(base < 2){ base = 2; } const counts = []; for(let i = 0; i < n; i++){ counts.push(base + (n - 1 - i) * gap); } let sum = counts.reduce((a,b) => a + b, 0); let diff = total - sum; // répartir le reste sans casser l'ordre strict while(diff > 0){ for(let i = 0; i < n && diff > 0; i++){ counts[i]++; diff--; } } // si on dépasse, retirer en partant du bas sans créer d'égalité while(diff < 0){ for(let i = n - 1; i >= 0 && diff < 0; i--){ const minAllowed = (i === n - 1) ? 2 : counts[i + 1] + 1; if(counts[i] > minAllowed){ counts[i]--; diff++; } } } // mélanger seulement les couleurs, pas les quantités const shuffledColors = Array.from({ length: n }, (_, i) => i) .sort(() => Math.random() - 0.5); const finalCounts = Array(n).fill(0); counts.forEach((count, i) => { finalCounts[shuffledColors[i]] = count; }); return finalCounts; }
