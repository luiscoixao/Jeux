function removeColor(colorIndex){ const els = [...board.querySelectorAll(`.balloon[data-color-index="${colorIndex}"]`)] .filter(el => !el.classList.contains('fly')); if(!els.length) return; isResolving = true; board.classList.add('resolving'); if( state.balloons.filter(balloon => balloon.colorIndex !== colorIndex).length === 0 && !state.seriesExpired && !state.gameExpired && performance.now() <= seriesDeadline && performance.now() <= gameDeadline ){ state.seriesWon = true; } let remaining = els.length; els.forEach((el) => { el.classList.add('fly'); const done = () => { el.removeEventListener('animationend', done); const id = Number(el.dataset.id); state.balloons = state.balloons.filter(balloon => balloon.id !== id); if(el.isConnected) el.remove(); remaining--; if(remaining === 0){ if(state.gameExpired){ finishGame(); }else if(state.balloons.length === 0){ advanceSeriesOrFinish(state.seriesWon); }else if(state.seriesExpired){ advanceSeriesOrFinish(false); }else{ isResolving = false; board.classList.remove('resolving'); render(); } } }; el.addEventListener('animationend', done, { once:true }); }); }
